 75class NALUnit:
 76    """Represents a RTP or H264 NAL unit
 77
 78    The structure of RTP NAL units are described in detail in [RFC 6184](https://www.rfc-editor.org/rfc/rfc6184)
 79    and H.264 NAL units are described in the [H.264 specification](https://www.itu.int/rec/T-REC-H.264/en)
 80    """
 81
 82    _START_CODE_PREFIX = b"\x00\x00\x01"
 83    _F_MASK = 0b10000000
 84    _F_SHIFT = 7
 85    _NRI_MASK = 0b01100000
 86    _NRI_SHIFT = 5
 87    _TYPE_MASK = 0b00011111
 88    _TYPE_SHIFT = 0
 89    _S_MASK = 0b10000000
 90    _S_SHIFT = 7
 91    _E_MASK = 0b01000000
 92    _E_SHIFT = 6
 93    _R_MASK = 0b00100000
 94    _R_SHIFT = 5
 95
 96    data: bytearray
 97    """Header and payload."""
 98
 99    def __init__(self, data: Union[bytes, bytearray, memoryview]) -> None:
100        self.data = bytearray(data)
101
102    @cached_property
103    def f(self) -> int:
104        """Forbidden zero bit."""
105        return (self.header & self._F_MASK) >> self._F_SHIFT
106
107    @cached_property
108    def nri(self) -> int:
109        """NAL ref IDC."""
110        return (self.header & self._NRI_MASK) >> self._NRI_SHIFT
111
112    @cached_property
113    def type(self) -> int:
114        """NAL unit type."""
115        return (self.header & self._TYPE_MASK) >> self._TYPE_SHIFT
116
117    @cached_property
118    def header(self) -> int:
119        """The header of the NAL unit or FU indicator in the case of a fragmentation unit."""
120        return self.data[0]
121
122    @property
123    def payload(self) -> bytes:
124        """The payload of the NAL unit."""
125        if isinstance(self, FUA):
126            return self.data[2:]
127        return self.data[1:]
128
129    @property
130    def data_with_prefix(self) -> bytes:
131        """The header and payload of the NAL unit with start code prefix prepended."""
132        return self._START_CODE_PREFIX + self.data
133
134    @classmethod
135    def from_rtp_payload(cls, rtp_payload: bytes) -> NALUnit:
136        """Constructs `NALUnit` from an rtp payload."""
137        nal_unit = cls(rtp_payload)
138        if nal_unit.type == 28:
139            return FUA(rtp_payload)
140        return nal_unit
141
142    @classmethod
143    def from_fu_a(cls, fu_a: FUA) -> NALUnit:
144        """Constructs `NALUnit` from an FUA.
145
146        Note that fragmented NAL unit payloads must be aggregated before they can get parsed.
147        """
148        header = fu_a.header & (cls._F_MASK | cls._NRI_MASK) | fu_a.original_type
149        data = bytearray()
150        data.append(header)
151        data += fu_a.payload
152        return cls(data)

155class FUA(NALUnit):
156    """A specific type of RTP NAL unit called FU-A (Fragmentation Unit type A).
157    Described in detail in RFC 6184 section [5.8](https://datatracker.ietf.org/doc/html/rfc6184#section-5.8).
158    """
159
160    @cached_property
161    def s(self) -> int:
162        """Start bit for fragmentation unit."""
163        return (self.fu_header & self._S_MASK) >> self._S_SHIFT
164
165    @cached_property
166    def e(self) -> int:
167        """End bit for fragmentation unit."""
168        return (self.fu_header & self._E_MASK) >> self._E_SHIFT
169
170    @cached_property
171    def original_type(self) -> int:
172        """The type of the NAL unit contained in the fragmentation unit."""
173        return (self.fu_header & self._TYPE_MASK) >> self._TYPE_SHIFT
174
175    @cached_property
176    def fu_header(self) -> int:
177        """The extra header in fragmentation units."""
178        return self.data[1]

181class Stream(RTPTransportClient, ABC):
182    """Abstract class for a RTSP media stream."""
183
184    transport: RTPTransport
185    """A wrapper around a pair of UDP (or TCP) sockets."""
186    rtp_queue: asyncio.Queue[Tuple[RTP, Optional[float]]]
187    """The queue where all received raw RTP packets get queued for demuxing and decoding."""
188    rtcp_queue: asyncio.Queue[RTCP]
189    """The queue where all received raw RTCP packets get queued for demuxing and decoding."""
190    type: StreamType
191    """The type of this media stream. For example scene camera or gaze."""
192    _last_rtcp_timestamp: Optional[int]
193    _last_ntp_time: Optional[float]
194
195    def __init__(self, transport: RTPTransport, type: StreamType) -> None:
196        transport.subscribe(self)
197        self.transport = transport
198        self.rtp_queue = asyncio.Queue()
199        self.rtcp_queue = asyncio.Queue(RTCP_QUEUE_SIZE)
200        self.type = type
201        self._last_rtcp_timestamp = None
202        self._last_ntp_time = None
203
204    def handle_rtp(self, rtp: RTP) -> None:
205        """A callback which is called everytime a new RTP packet is received. Queues the packet and
206        calculates its absolute NTP timestamp."""
207        if self._last_ntp_time is not None and self._last_rtcp_timestamp is not None:
208            time_delta = cast(int, rtp.ts) - self._last_rtcp_timestamp  # type: ignore
209            ntp_timestamp = self._last_ntp_time + time_delta / TIMESTAMP_GRANULARITY
210        else:
211            ntp_timestamp = None
212        self.rtp_queue.put_nowait((rtp, ntp_timestamp))
213        # _logger.debug(f"{self.type}: {rtp.ts}")
214        # _logger.debug(f"RTP size: {len(rtp.data)}")
215
216    def handle_rtcp(self, rtcp: RTCP) -> None:
217        """A callback which is called everytime a new RTCP packet is received. Queues the packet and
218        extracts information needed for calculations of absolute time."""
219        try:
220            self.rtcp_queue.put_nowait(rtcp)
221        except asyncio.QueueFull:
222            _logger.warning(
223                "RTCP queue full. New RTCP packages will be thrown away. Consume the queue to prevent this from happening."
224            )
225        sender_report = cast(Optional[SR], rtcp.get(200))
226        if sender_report is None:
227            return
228        self._last_ntp_time = sender_report.ntp
229        self._last_rtcp_timestamp = sender_report.ts

313    async def demux(
314        self,
315    ) -> AsyncIterator[asyncio.Queue[Tuple[bytes, Optional[float]]]]:
316        data_queue: asyncio.Queue[Tuple[bytes, Optional[float]]] = asyncio.Queue(
317            DATA_QUEUE_SIZE
318        )
319
320        async def demuxer():
321            while True:
322                rtp, timestamp = await self.rtp_queue.get()

328        finally:

332            except asyncio.CancelledError:

336    async def decode(
337        self,
338    ) -> AsyncIterator[asyncio.Queue[Tuple[JSONObject, Optional[float]]]]:
339        json_queue: asyncio.Queue[Tuple[JSONObject, Optional[float]]] = asyncio.Queue(
340            DATA_QUEUE_SIZE
341        )
342
343        async def decoder():
344            async with self.demux() as data_queue:
345                while True:

352                        )
353                        continue
354                    await json_queue.put((json_message, timestamp))
355
356        decoder_task = _utils.create_task(decoder(), name="decoder")
357        try:
358            yield json_queue
359        finally:
360            decoder_task.cancel()
361            try:
362                await decoder_task

367class VideoStream(Stream):
368    """Represents a RTSP video stream.
369
370    Handles demuxing and decoding of video frames.
371    """
372
373    _nal_unit_builder: Tuple[NALUnit, Optional[float]]
374
375    def __init__(self, transport: RTPTransport, stream_type: StreamType) -> None:
376        super().__init__(transport, stream_type)
377        self.codec_context: Any = av.CodecContext.create("h264", "r")  # type: ignore
378        self.sps_or_pps_received = False
379        self._demux_in_count = 0
380        self._demux_out_count = 0
381        self._decode_count = 0
382        self._fragment_count = 0
383
384    @property
385    def media_type(self) -> MediaType:
386        """The media type identifier of a `VideoStream`."""
387        return "video"
388
389    @property
390    def stats(self) -> Dict[str, int]:
391        """Contains some media stream statistics. Used mainly for debugging purposes."""
392        return {
393            "demux_in_count": self._demux_in_count,
394            "demux_out_count": self._demux_out_count,
395            "decode_count": self._decode_count,
396        }
397
398    @asynccontextmanager
399    async def demux(
400        self,
401    ) -> AsyncIterator[asyncio.Queue[Tuple[NALUnit, Optional[float]]]]:
402        """Returns a queue with tuples containing the demuxed RTP stream along with timestamps.
403
404        Spawns a demuxer task which parses the NAL units received in the RTP payloads.
405        It also aggregates fragmentation units of larger NAL units sent in multiple RTP packets.
406        """
407        nal_unit_queue: asyncio.Queue[Tuple[NALUnit, Optional[float]]] = asyncio.Queue(
408            FRAME_QUEUE_SIZE
409        )
410
411        async def demuxer():
412            while True:
413                rtp, timestamp = await self.rtp_queue.get()
414                self._demux_in_count += 1
415                # t0 = time.perf_counter()
416                nal_unit = NALUnit.from_rtp_payload(cast(bytes, rtp.data))  # type: ignore
417                if nal_unit.type in [7, 8]:
418                    # SPS or PPS is queued
419                    self.sps_or_pps_received = True
420                    await nal_unit_queue.put((nal_unit, timestamp))
421                    self._demux_out_count += 1
422                    continue
423                if not self.sps_or_pps_received:
424                    # SPS or PPS should be the first NAL unit to be queued
425                    continue
426                if nal_unit.type in [1, 5]:
427                    # Self contained NAL unit is queued
428                    await nal_unit_queue.put((nal_unit, timestamp))
429                    self._demux_out_count += 1
430                    continue
431                if isinstance(nal_unit, FUA):
432                    # Fragmented NAL units need to be aggregated
433                    if nal_unit.s:
434                        self._nal_unit_builder = (
435                            NALUnit.from_fu_a(nal_unit),
436                            timestamp,
437                        )
438                        self._fragment_count = 1
439                        # t1 = time.perf_counter()
440                        # logger.debug(f"Demuxed FU-A start in {t1 - t0:.6f} seconds")
441                        continue
442                    self._nal_unit_builder[0].data += nal_unit.payload
443                    self._fragment_count += 1
444                    # t1 = time.perf_counter()
445                    # logger.debug(f"Demuxed FU-A in {t1 - t0:.6f} seconds")
446                    if nal_unit.e:
447                        # logger.debug(f"NAL unit built of {self._fragment_count} FU-As")
448                        await nal_unit_queue.put(self._nal_unit_builder)
449                        self._demux_out_count += 1
450                else:
451                    _logger.warning(f"Unhandled NAL unit of type {nal_unit.type}")
452
453        demuxer_task = _utils.create_task(demuxer(), name="demuxer")
454        try:
455            yield nal_unit_queue
456        finally:
457            demuxer_task.cancel()
458            try:
459                await demuxer_task
460            except asyncio.CancelledError:
461                pass
462
463    @asynccontextmanager
464    async def decode(self) -> AsyncIterator[asyncio.Queue[Tuple[Any, Optional[float]]]]:
465        """Returns a queue with tuples containing the demuxed and decoded RTP stream along with timestamps.
466
467        Spawns a decoder task which uses PyAV (ffmpeg) to parse and decode the demuxed NAL units.
468
469        The returned queue contains PyAVs `av.VideoFrame` objects along with timestamps.
470        """
471        frame_queue: asyncio.Queue[Tuple[Any, Optional[float]]] = asyncio.Queue(
472            FRAME_QUEUE_SIZE
473        )
474
475        async def decoder():
476            async with self.demux() as nal_unit_queue:
477                while True:
478                    nal_unit, timestamp = await nal_unit_queue.get()
479                    # t0 = time.perf_counter()
480                    packets = cast(
481                        List[Any],
482                        self.codec_context.parse(nal_unit.data_with_prefix),
483                    )
484                    frames = functools.reduce(
485                        lambda frame_acc, packet: frame_acc
486                        + self.codec_context.decode(packet),
487                        packets,
488                        [],
489                    )
490                    # t1 = time.perf_counter()
491                    # logger.debug(f"Decoded NAL unit in {t1 - t0:.6f} seconds")
492                    for frame in frames:
493                        await frame_queue.put((frame, timestamp))
494                        self._decode_count += 1
495
496        decoder_task = _utils.create_task(decoder(), name="decoder")
497        try:
498            yield frame_queue
499        finally:
500            decoder_task.cancel()
501            try:
502                await decoder_task
503            except asyncio.CancelledError:
504                pass

507class Streams:
508    """Handles a `RTSPMediaSession` with one or multiple media streams.
509
510    Exposes a `connect` function which is used to set up an RTSP media session and create an instance of this object.
511    Gives easy access to the different streams.
512
513    After the setup process is completed, await the `play` coroutine to start the streaming.
514    """
515
516    def __init__(self, session: RTSPMediaSession, streams: Set[Stream]) -> None:
517        self.session = session
518        self.streams: Dict[StreamType, Stream] = {
519            stream.type: stream for stream in streams
520        }
521
522    @property
523    def scene_camera(self) -> VideoStream:
524        return cast(VideoStream, self._get_stream(StreamType.SCENE_CAMERA))
525
526    @property
527    def audio(self) -> Stream:
528        return self._get_stream(StreamType.AUDIO)
529
530    @property
531    def eye_cameras(self) -> Stream:
532        return self._get_stream(StreamType.EYE_CAMERAS)
533
534    @property
535    def gaze(self) -> Stream:
536        return self._get_stream(StreamType.GAZE)
537
538    @property
539    def sync(self) -> Stream:
540        return self._get_stream(StreamType.SYNC)
541
542    @property
543    def imu(self) -> Stream:
544        return self._get_stream(StreamType.IMU)
545
546    @property
547    def events(self) -> Stream:
548        return self._get_stream(StreamType.EVENTS)
549
550    def _get_stream(self, stream_type: StreamType) -> Stream:
551        try:
552            return self.streams[stream_type]
553        except KeyError:
554            raise AttributeError(
555                f"The {stream_type.name} stream was never initialized.",
556                name=stream_type.property_name,  # type: ignore
557                obj=self,  # type: ignore
558            )
559
560    @classmethod
561    @asynccontextmanager
562    async def connect(
563        cls,
564        rtsp_url: str,
565        scene_camera: bool = True,
566        audio: bool = False,
567        eye_cameras: bool = False,
568        gaze: bool = False,
569        sync: bool = False,
570        imu: bool = False,
571        events: bool = False,
572    ) -> AsyncIterator[Streams]:
573        """Sets up an RTSP media session with the specified streams and creates an instance of `Streams`."""
574        parsed_url = urlparse(rtsp_url)
575        async with RTSPConnection(parsed_url.hostname, parsed_url.port) as connection:
576            async with AsyncExitStack() as stack:
577                streams: Set[Stream] = set()
578                if scene_camera:
579                    streams.add(
580                        await stack.enter_async_context(
581                            VideoStream.setup(
582                                connection, StreamType.SCENE_CAMERA, parsed_url.scheme
583                            )
584                        )
585                    )
586                if eye_cameras:
587                    streams.add(
588                        await stack.enter_async_context(
589                            VideoStream.setup(
590                                connection, StreamType.EYE_CAMERAS, parsed_url.scheme
591                            )
592                        )
593                    )
594                if gaze:
595                    streams.add(
596                        await stack.enter_async_context(
597                            DataStream.setup(
598                                connection, StreamType.GAZE, parsed_url.scheme
599                            )
600                        )
601                    )
602                if audio or sync or imu or events:
603                    raise NotImplementedError()
604
605                async with RTSPMediaSession(
606                    connection,
607                    rtsp_url,
608                    media_stream_configurations=list(
609                        map(lambda s: s.media_stream_configuration, streams)
610                    ),
611                ) as session:
612                    yield cls(session, streams)
613

615        """Starts the streaming in the RTSP media session."""
616        await self.session.play()  # type: ignore

401        self,
402    ) -> AsyncIterator[asyncio.Queue[Tuple[NALUnit, Optional[float]]]]:
403        """Returns a queue with tuples containing the demuxed RTP stream along with timestamps.
404
405        Spawns a demuxer task which parses the NAL units received in the RTP payloads.
406        It also aggregates fragmentation units of larger NAL units sent in multiple RTP packets.
407        """
408        nal_unit_queue: asyncio.Queue[Tuple[NALUnit, Optional[float]]] = asyncio.Queue(
409            FRAME_QUEUE_SIZE
410        )
411
412        async def demuxer():
413            while True:
414                rtp, timestamp = await self.rtp_queue.get()
415                self._demux_in_count += 1
416                # t0 = time.perf_counter()
417                nal_unit = NALUnit.from_rtp_payload(cast(bytes, rtp.data))  # type: ignore
418                if nal_unit.type in [7, 8]:
419                    # SPS or PPS is queued
420                    self.sps_or_pps_received = True
421                    await nal_unit_queue.put((nal_unit, timestamp))
422                    self._demux_out_count += 1
423                    continue
424                if not self.sps_or_pps_received:
425                    # SPS or PPS should be the first NAL unit to be queued
426                    continue
427                if nal_unit.type in [1, 5]:
428                    # Self contained NAL unit is queued
429                    await nal_unit_queue.put((nal_unit, timestamp))
430                    self._demux_out_count += 1
431                    continue
432                if isinstance(nal_unit, FUA):
433                    # Fragmented NAL units need to be aggregated
434                    if nal_unit.s:
435                        self._nal_unit_builder = (
436                            NALUnit.from_fu_a(nal_unit),
437                            timestamp,
438                        )
439                        self._fragment_count = 1
440                        # t1 = time.perf_counter()
441                        # logger.debug(f"Demuxed FU-A start in {t1 - t0:.6f} seconds")
442                        continue
443                    self._nal_unit_builder[0].data += nal_unit.payload
444                    self._fragment_count += 1
445                    # t1 = time.perf_counter()
446                    # logger.debug(f"Demuxed FU-A in {t1 - t0:.6f} seconds")
447                    if nal_unit.e:
448                        # logger.debug(f"NAL unit built of {self._fragment_count} FU-As")
449                        await nal_unit_queue.put(self._nal_unit_builder)
450                        self._demux_out_count += 1
451                else:
452                    _logger.warning(f"Unhandled NAL unit of type {nal_unit.type}")
453
454        demuxer_task = _utils.create_task(demuxer(), name="demuxer")
455        try:
456            yield nal_unit_queue
457        finally:
458            demuxer_task.cancel()
459            try:

401        self,
402    ) -> AsyncIterator[asyncio.Queue[Tuple[NALUnit, Optional[float]]]]:
403        """Returns a queue with tuples containing the demuxed RTP stream along with timestamps.
404
405        Spawns a demuxer task which parses the NAL units received in the RTP payloads.
406        It also aggregates fragmentation units of larger NAL units sent in multiple RTP packets.
407        """
408        nal_unit_queue: asyncio.Queue[Tuple[NALUnit, Optional[float]]] = asyncio.Queue(
409            FRAME_QUEUE_SIZE
410        )
411
412        async def demuxer():
413            while True:
414                rtp, timestamp = await self.rtp_queue.get()
415                self._demux_in_count += 1
416                # t0 = time.perf_counter()
417                nal_unit = NALUnit.from_rtp_payload(cast(bytes, rtp.data))  # type: ignore
418                if nal_unit.type in [7, 8]:
419                    # SPS or PPS is queued
420                    self.sps_or_pps_received = True
421                    await nal_unit_queue.put((nal_unit, timestamp))
422                    self._demux_out_count += 1
423                    continue
424                if not self.sps_or_pps_received:
425                    # SPS or PPS should be the first NAL unit to be queued
426                    continue
427                if nal_unit.type in [1, 5]:
428                    # Self contained NAL unit is queued
429                    await nal_unit_queue.put((nal_unit, timestamp))
430                    self._demux_out_count += 1
431                    continue
432                if isinstance(nal_unit, FUA):
433                    # Fragmented NAL units need to be aggregated
434                    if nal_unit.s:
435                        self._nal_unit_builder = (
436                            NALUnit.from_fu_a(nal_unit),
437                            timestamp,
438                        )
439                        self._fragment_count = 1
440                        # t1 = time.perf_counter()
441                        # logger.debug(f"Demuxed FU-A start in {t1 - t0:.6f} seconds")
442                        continue
443                    self._nal_unit_builder[0].data += nal_unit.payload
444                    self._fragment_count += 1
445                    # t1 = time.perf_counter()
446                    # logger.debug(f"Demuxed FU-A in {t1 - t0:.6f} seconds")
447                    if nal_unit.e:
448                        # logger.debug(f"NAL unit built of {self._fragment_count} FU-As")
449                        await nal_unit_queue.put(self._nal_unit_builder)
450                        self._demux_out_count += 1
451                else:
452                    _logger.warning(f"Unhandled NAL unit of type {nal_unit.type}")
453
454        demuxer_task = _utils.create_task(demuxer(), name="demuxer")
455        try:
456            yield nal_unit_queue
457        finally:
458            demuxer_task.cancel()
459            try:
460                await demuxer_task
461            except asyncio.CancelledError:
462                pass
